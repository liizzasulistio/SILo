\cf0 CREATE TABLE users\
(\
	UserID INT AUTO_INCREMENT PRIMARY KEY,\
	UserName VARCHAR(50),\
	UserType ENUM('Booker', 'Warehouse', 'Cashier'),\
	UserEmail VARCHAR(50),\
	UserPassword VARCHAR(255)\
)\
\
CREATE TABLE items\
(\
	ItemID INT AUTO_INCREMENT PRIMARY KEY,\
	ItemBarcode VARCHAR(50),\
	ItemTitle VARCHAR(50),\
	ItemDescription VARCHAR(50),\
	ItemManufacturer VARCHAR(50),\
	ItemURL VARCHAR(50),\
	ItemStock INT\
)\
\
CREATE TABLE purchase_orders\
(\
	PurchaseOrderNumber INT AUTO_INCREMENT PRIMARY KEY,\
	PurchaseOrderDate DATE\
)\
\
CREATE TABLE order_details\
(\
	OrderID INT AUTO_INCREMENT PRIMARY KEY,\
	PurchaseOrderNumber_FK INT,\
	ItemID_FK INT,\
	ItemQuantity INT,\
	FOREIGN KEY (PurchaseOrderNumber_FK) REFERENCES purchase_orders(PurchaseOrderNumber) ON UPDATE CASCADE,\
	FOREIGN KEY (ItemID_FK) REFERENCES items(ItemID) ON UPDATE CASCADE\
)\
\
CREATE TABLE invoices\
(\
	InvoiceNumber INT AUTO_INCREMENT PRIMARY KEY,\
	PurchaseOrderNumber_FK INT,\
	InvoiceSupplierName VARCHAR(50),\
	InvoiceStatus VARCHAR(50),\
	FOREIGN KEY (PurchaseOrderNumber_FK) REFERENCES purchase_orders(PurchaseOrderNumber) ON UPDATE CASCADE\
)\
\
CREATE TABLE customers\
(\
	CustomerID INT AUTO_INCREMENT PRIMARY KEY,\
	CustomerName VARCHAR(50),\
	CustomerEmailAddress VARCHAR(50)\
)\
\
CREATE TABLE delivery_notes\
(\
	DeliveryNoteNumber INT AUTO_INCREMENT PRIMARY KEY,\
	DeliveryNoteStatus VARCHAR(50),\
	DeliveryNoteDescription VARCHAR(50),\
	CustomerID_FK INT,\
	InvoiceNumber_FK INT,\
	PurchaseOrderNumber_FK INT,\
	FOREIGN KEY (CustomerID_FK) REFERENCES customers(CustomerID) ON UPDATE CASCADE,\
	FOREIGN KEY (InvoiceNumber_FK) REFERENCES invoices(InvoiceNumber) ON UPDATE CASCADE,\
	FOREIGN KEY (PurchaseOrderNumber_FK) REFERENCES purchase_orders(PurchaseOrderNumber) ON UPDATE CASCADE\
)\
\
CREATE TABLE packages\
(\
	PackageNumber INT AUTO_INCREMENT PRIMARY KEY,\
	InvoiceNumber_FK INT,\
	FOREIGN KEY (InvoiceNumber_FK) REFERENCES invoices(InvoiceNumber) ON UPDATE CASCADE\
)}
